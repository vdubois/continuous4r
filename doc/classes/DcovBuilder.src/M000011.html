  def quality_percentage
    require 'hpricot'
    doc = Hpricot(File.read("#{Continuous4r::WORK_DIR}/dcov/coverage.html"))
    doc.search('//h3') do |h3|
      if h3.inner_text.match(/^Global coverage percentage/)
        return h3.inner_text.split(/Global coverage percentage : /)[1].split(/%/)[0]
      end
    end
  end
